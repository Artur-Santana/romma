  async function fetchDados() {
    // setState calls dentro do body do effect (não indireção externa)
  }
  fetchDados()
}, [])

Por que funciona: a regra dispara quando setState é chamado via função definida FORA do escopo do useEffect.
Declarar a async dentro do body coloca o setState no escopo local do effect — regra não dispara.
